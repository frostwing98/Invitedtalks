\section{Background}

\frame{
\frametitle{Initial attempt: Protocol}

\begin{itemize}
  \item Traditional crypto(802.11i RSNA)
  \item Management is cumbersome
  \item Signal strengtt: the only criteria
  \item Same ID, different channel
\end{itemize}
}
\frame{
\frametitle{other ways}
\begin{itemize}
  \item MAC whitelist
  \item RSS values
  \item clock skews
  \item radio frequency variations
\end{itemize}
}
\frame{
\frametitle{Follow-up attemps: timestamp based, mobicom'08}
\begin{itemize}
  \item Use TCP/IP Timestamp(X)
  \item Use TSF(Time
Synchronization Function) Timestamp
\begin{itemize}
  \item high granularity: 1ms
  \item timestamp is actual time: no need for considering delays
\end{itemize}
\end{itemize}

}
\frame{
\frametitle{Follow-up attemps: timestamp based, mobicom'08}
Fitting the timestamp patterns
\begin{itemize}
  \item Linear programming method (LPM) to avoid delay
  \item Least Square Fitting (LSF) to estimate
  \item "Jump" exists: different AP
\end{itemize}
}
\frame{
\frametitle{Follow-up attemps: RTT-based, infocom'09}
Rogue AP: 2 interfaces:
\begin{itemize}
  \item Interface connecting to legitimate AP
  \item Interface pretending to be legal
\end{itemize}
Adversarial Examples:
\begin{itemize}
  \item identity faking: easy to change
  \item traffic monitoring: both are wireless
  \item simple timing or RTT(round trip time): just create a response
\end{itemize}
}
\frame{
\frametitle{Follow-up attemps: RTT-based, infocom'09}
RTT time of DNS-lookup non-recursive query
\begin{itemize}
  \item most local networks has a local DNS server
  \item a station can always send a dns-lookup request
  \item DNS request cannot be blocked by network providers
  \item DNS response varies for different queries
  \item unpredictable, rogue AP forced to forward request.
  \item Always "no such host name": rogue AP
\end{itemize}
}
